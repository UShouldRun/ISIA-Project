 93        Returns:

 95        """
 96        grid = []
 97        terrain_choices = [-1, 0, 1]
 98        weights = [0.05, 0.90, 0.05]
 99
100        for i in range(self.columns):
101            column = []
102            for j in range(self.rows):
103                initial_terrain = random.choices(terrain_choices, weights=weights, k=1)[0]
104                cell = MapCell(pos=(i, j), terrain=initial_terrain)
105                column.append(cell)
106            grid.append(column)

325class AStar:
326    """Static class implementing the A* pathfinding algorithm."""
327
328    @staticmethod
329    def _reconstruct(map: Map, path: dict[MapPos, MapPos], start: MapPos, goal: MapPos) -> List[Tuple[float, float]]:
330        """
331        Reconstruct the path from start to goal.
332
333        Args:
334            map (Map): The map object.
335            path (dict): Parent pointers for each node.
336            start (MapPos): Start node.
337            goal (MapPos): Goal node.
338
339        Returns:
340            List[Tuple[float, float]]: List of physical coordinates representing the path.
341        """
342        node: MapPos = goal
343        seq: List[MapPos] = []
344
345        while node != start:
346            seq.insert(0, map.rescale(node))
347            node = path[node]
348
349        return seq
350
351    @staticmethod
352    def _gScore(map: Map, curr: MapPos, neighbour: MapPos) -> float:
353        """
354        Compute the cost to move from curr to neighbour.
355
356        Args:
357            map (Map): Map object.
358            curr (MapPos): Current node.
359            neighbour (MapPos): Neighbour node.
360
361        Returns:
362            float: Movement cost.
363        """
364        return map.distance(curr, neighbour) + map.grid[neighbour[0]][neighbour[1]].get_cost()
365
366    @staticmethod
367    def _heuristicScore(map: Map, curr: MapPos, goal: MapPos) -> float:
368        """
369        Heuristic estimate from curr to goal.
370
371        Args:
372            map (Map): Map object.
373            curr (MapPos): Current node.
374            goal (MapPos): Goal node.
375
376        Returns:
377            float: Heuristic cost.
378        """
379        return map.distance(curr, goal)
380
381    @staticmethod
382    def run(map: Map, start: Tuple[float, float], goal: Tuple[float, float]) -> List[Tuple[float, float]]:
383        """
384        Run the A* algorithm to find a path from start to goal.
385
386        Args:
387            map (Map): Map object.
388            start (Tuple[float, float]): Start coordinates.
389            goal (Tuple[float, float]): Goal coordinates.
390
391        Returns:
392            List[Tuple[float, float]]: List of physical coordinates representing the path.
393        """
394        print(f"{YELLOW}AStar[STARTING]: start = {start}, goal = {goal}{RESET}")
395
396        s: MapPos = map.normalize(start)
397        g: MapPos = map.normalize(goal)
398
399        min_heap: List[AStarNode] = []
400        heapify(min_heap)
401
402        path: dict[MapPos, MapPos] = {}
403        gScore: dict[MapPos, float] = defaultdict(lambda: float('inf'))
404        gScore[s] = 0
405        heappush(min_heap, AStarNode(s, gScore[s]))
406
407        fScore: dict[MapPos, float] = defaultdict(lambda: float('inf'))
408        fScore[s] = AStar._heuristicScore(map, s, g)
409
410        count: int = 0
411        while min_heap:
412            curr: MapPos = heappop(min_heap).pos
413            if map.is_visited(curr):
414                continue
415            map.visit(curr)
416            count += 1
417
418            if curr == g:
419                print(f"{YELLOW}AStar[FINISHING]: reconstructing path{RESET}")
420                map.clear_visited()
421                return AStar._reconstruct(map, path, s, g)
422
423            neighbours: List[MapPos] = filter(
424                lambda pos: map.in_map(pos) and not map.is_visited(pos),
425                [(curr[0] + dx, curr[1] + dy)
426                 for dx, dy in [(-1, -1), (0, -1), (1, -1), (1, 0),
427                                 (1, 1), (0, 1), (-1, 1), (-1, 0)]]
428            )
429
430            for neighbour in neighbours:
431                tentative_gScore: float = gScore[curr] + AStar._gScore(map, curr, neighbour)
432
433                if tentative_gScore < gScore[neighbour]:
434                    path[neighbour] = curr
435                    gScore[neighbour] = tentative_gScore
436                    fScore[neighbour] = tentative_gScore + AStar._heuristicScore(map, neighbour, g)
437                    heappush(min_heap, AStarNode(neighbour, fScore[neighbour]))
438
439        print(f"{YELLOW}AStar[FINISHING]: did not find path{RESET}")
440        map.clear_visited()
441        return []
