109    def get_dpos(self, curr: Tuple[float, float], next: Tuple[float, float]) -> Tuple[int, int]:
110        """
111        Compute the directional delta between two points.
112
113        Args:
114            curr (Tuple[float, float]): Current position.
115            next (Tuple[float, float]): Target position.
116
117        Returns:
118            Tuple[int, int]: Delta vector.
119        """
120        return (next[0] - curr[0], next[1] - curr[1])

122    async def try_go_around(self, goal: Tuple[float, float]) -> Optional[Tuple[float, float]]:
123        """
124        Attempt simple local obstacle avoidance using random offsets.
125
126        Args:
127            goal (Tuple[float, float]): Target waypoint.
128
129        Returns:
130            Optional[Tuple[float, float]]: New temporary position or None.
131        """
132        for _ in range(5):
133            offset_x = random.uniform(-0.5 * ROVER_SPEED_UNIT_PER_SEC, 0.5 * ROVER_SPEED_UNIT_PER_SEC)
134            offset_y = random.uniform(-0.5 * ROVER_SPEED_UNIT_PER_SEC, 0.5 * ROVER_SPEED_UNIT_PER_SEC)
135            candidate = (self.position[0] + offset_x, self.position[1] + offset_y)
136            if not self.world.collides(self.jid, candidate):
137                print(f"{CYAN}[{self.name}] Avoiding obstacle locally  {candidate}{RESET}")
138                return candidate
139        return None

141    async def find_path(self) -> str:
142        """
143        Compute an A* path to the current rover goal.
144
145        Returns:
146            str: Status code indicating result ("viable", "no_path", "not_enough_energy").
147        """
148        rover = self
149
150        rover.path = AStar.run(rover.map, rover.position, rover.goal)
151        if not rover.path:
152            print(f"{CYAN}[{rover.name}] Did not find path to the goal, rejecting mission{RESET}")
153            rover.status = "idle"
154            rover.goal = None
155            return "no_path"
156        print(f"{CYAN}[{rover.name}] Found path to the goal{RESET}")
157
158        distance = sum(
159            rover.calculate_distance(rover.path[i - 1], rover.path[i])
160            for i in range(1, len(rover.path))
161        )
162        if rover.energy < ((2.2 + 0.10 * STORM_CHANCE * STORM_COST) * distance) * ENERGY_PER_DISTANCE_UNIT:
163            print(f"{CYAN}[{rover.name}] Low on energy; rejecting mission to {rover.goal}{RESET}")
164            rover.status = "idle"
165            rover.goal = None
166            rover.path = []
167            return "not_enough_energy"
168
169        print(f"{CYAN}[{rover.name}] Found path to the goal and has enough energy; accepting mission to {rover.goal}{RESET}")
170        rover.status = "moving"
171        return "viable"

173    def calculate_distance(self, pos1: Tuple[float, float], pos2: Tuple[float, float]) -> float:
174        """
175        Compute Euclidean distance between two points.
176
177        Args:
178            pos1 (Tuple[float, float]): First point.
179            pos2 (Tuple[float, float]): Second point.
180
181        Returns:
182            float: Euclidean distance.
183        """
184        return sqrt((pos1[0] - pos2[0]) ** 2 + (pos1[1] - pos2[1]) ** 2)

186    def calculate_pathfinding_cost(self, start_pos: Tuple[float, float], target_pos: Tuple[float, float]) -> float:
187        """
188        Estimate pathfinding cost using randomized Euclidean distance.
189
190        Args:
191            start_pos (Tuple[float, float]): Starting point.
192            target_pos (Tuple[float, float]): Target point.
193
194        Returns:
195            float: Cost estimate.
196        """
197        euclidean_dist = self.calculate_distance(start_pos, target_pos)
198        return euclidean_dist * random.uniform(1.0, 1.1)

200    def compute_mission_time(self, target_pos: Tuple[float, float]) -> float:
201        """
202        Compute estimated total mission time for bidding.
203
204        Args:
205            target_pos (Tuple[float, float]): Mission target.
206
207        Returns:
208            float: Estimated mission time in seconds.
209        """
210        current_energy = self.energy
211
212        dist_to_target = self.calculate_pathfinding_cost(self.position, target_pos)
213        dist_to_base_after = self.calculate_pathfinding_cost(target_pos, self.position)
214        total_distance = dist_to_target + dist_to_base_after
215
216        energy_required = total_distance * ENERGY_PER_DISTANCE_UNIT
217
218        time_to_charge = 0.0
219        if current_energy < energy_required:
220            energy_needed_to_charge = energy_required - current_energy
221            time_to_charge = energy_needed_to_charge / CHARGE_RATE_ENERGY_PER_SEC
222
223        time_to_travel = total_distance / ROVER_SPEED_UNIT_PER_SEC
224
225        mission_time = time_to_charge + time_to_travel
226
227        return mission_time

273    class ReceiveMessages(CyclicBehaviour):
274        """
275        Behaviour responsible for receiving and handling incoming XMPP messages.
276
277        The behaviour processes bid requests from the base, sends proposals or
278        refusals, accepts/rejects mission confirmations, and updates rover
279        state based on incoming informs. It communicates with the base using
280        structured Message objects with metadata and a stringified body.
281        """
282        async def run(self):
283            """
284            Wait for messages with a short timeout and handle several
285            performatives:
286
287            - cfp / rover_bid_cfp: evaluate mission and send a proposal or refuse
288            - accept_proposal: start moving toward the accepted target
289            - reject_proposal: unlock and remain idle
290            - inform (return_path_to_base): receive a return path and update state
291
292            The method uses eval() to parse message bodies that were serialized
293            with str(dict). It also forwards human-readable updates to the
294            visualizer.
295            """
296            rover = self.agent
297            msg = await self.receive(timeout=3)
298            if not msg:
299                return
300
301            performative = msg.metadata.get("performative")
302            msg_type = msg.metadata.get("type")
303            sender = str(msg.sender)
304
305            # --------------------------
306            # BASE  ROVER : Bid Request
307            # --------------------------
308            if performative == "cfp" and msg_type == "rover_bid_cfp":
309                target_pos = eval(msg.body)
310                print(f"{CYAN}[{rover.name}] Received Bid request from Base for {target_pos}{RESET}")
311                await rover.viz_send_message(f"Received mission bid request for {target_pos}")
312
313                # Check if rover is free
314                if rover.is_locked_by_bid or rover.status != "idle":
315                    # Already committed  refuse
316                    reply = Message(
317                        to=sender,
318                        metadata={"performative": "refuse", "type": "rover_bid_cfp"},
319                        body=str({"reason": "busy or locked"})
320                    )
321                    await self.send(reply)
322                    print(f"{CYAN}[{rover.name}] REFUSING mission at {target_pos} (busy/locked){RESET}")
323                    await rover.viz_send_message(f"Refused mission at {target_pos} (busy/locked)")
324
325                else:
326                    # Rover is available  propose
327                    rover.is_locked_by_bid = True  # lock it for this bid
328
329                    rover.goal = target_pos
330                    mission_status = await rover.find_path()
331                    if mission_status != "viable":
332                        rover.is_locked_by_bid = False
333                        reply = Message(
334                            to=sender,
335                            metadata={"performative": "refuse", "type": "rover_bid_cfp"},
336                            body=str({"reason": mission_status})
337                        )
338                        await self.send(reply)
339                        print(f"{CYAN}[{rover.name}] REFUSING mission at {target_pos} ({mission_status}){RESET}")
340                        await rover.viz_send_message(f"Refused mission at {target_pos} ({mission_status})")
341                        return
342
343                    estimated_mission_time = rover.compute_mission_time(target_pos)
344                    proposal = {"cost": estimated_mission_time, "rover": str(rover.jid)}
345                    reply = Message(
346                        to=sender,
347                        metadata={"performative": "propose", "type": "rover_bid_cfp"},
348                        body=str(proposal)
349                    )
350                    await self.send(reply)
351                    print(f"{CYAN}[{rover.name}] PROPOSING mission at {target_pos} with cost {estimated_mission_time:.2f}{RESET}")
352                    await rover.viz_send_message(f"Submitted bid for {target_pos} (cost: {estimated_mission_time:.1f}s)")
353
354            # Go to target
355            elif performative == "accept_proposal" and msg_type == "rover_bid_cfp":
356                rover.status = "moving"
357                rover.goal = eval(msg.body)["target"]
358                rover.is_locked_by_bid = False  # Unlock after acceptance
359                rover.is_on_base = False # Out of base
360
361                print(f"{CYAN}[{rover.name}] ACCEPTED mission to {rover.goal}{RESET}")
362                await rover.viz_send_message(f"Mission accepted! Moving to {rover.goal}")
363                rover.add_behaviour(rover.MoveAlongPath())
364
365            # Reject proposal  unlock and stay idle
366            elif performative == "reject_proposal" and msg_type == "rover_bid_cfp":
367                rover.is_locked_by_bid = False  # Unlock the rover
368                rover.status = "idle"
369                rover.goal = None
370                rover.path = []
371                print(f"{CYAN}[{rover.name}] REJECTED for mission at {eval(msg.body)['target']}{RESET}")
372                await rover.viz_send_message(f"Bid rejected for mission at {eval(msg.body)['target']}")
373
374            # --------------------------
375            # Inform on rover stats
376            # --------------------------
377            elif performative == "inform" and ontology == "return_path_to_base":
378                rover.path = eval(msg.body)
379                rover.goal = rover.path[-1] if rover.path else None
380                rover.status = "returning"
381                rover.is_locked_by_bid = False
382                print(f"[{rover.name}] Received return path ({len(rover.path)} steps) to base.")

384    class MoveAlongPath(CyclicBehaviour):
385        """
386        Behaviour that moves the rover along a previously computed path.
387
388        Responsibilities:
389        - Notify the base when leaving
390        - Step the rover position along the path according to move_step and

392        - Detect collisions and attempt local avoidance
393        - Trigger mission-complete events and start analysis behaviour
394        """
395        async def on_start(self) -> None:
396            """
397            Called when the behaviour is added to the agent. If the rover is in
398            the moving state it sends an inform message to the base announcing
399            departure.
400            """
401            rover = self.agent 
402
403            if rover.status == "moving":
404                print(f"{CYAN}[{rover.name}] Informing moving to goal to base{RESET}")
405                msg = Message(
406                    to=rover.base_jid,
407                    metadata={"performative": "inform", "type": "rover_leaving_base"},
408                    body=str({"goal": rover.goal})
409                )
410                await self.send(msg)

412        async def run(self):
413            """
414            Execute a single movement step along the rover's path. The method
415            performs checks for valid goal/state/path, computes the next
416            position, updates energy, handles collisions, and triggers
417            subsequent behaviours when the rover arrives at its destination or
418            returns to base.
419            """
420            rover = self.agent
421
422            if rover.goal == None:
423                print(f"{CYAN}[{rover.name}] No goal set, cancelling MoveAlongPath{RESET}")
424                self.kill()
425                await asyncio.sleep(2)
426                return
427
428            if rover.status not in ["moving", "returning"]:
429                print(f"{CYAN}[{rover.name}] Doing another task, cancelling MoveAlongPath{RESET}")
430                self.kill()
431                await asyncio.sleep(2)
432                return
433
434            if not rover.path:
435                print(f"{CYAN}[{rover.name}] could not compute path to goal {rover.goal}{RESET}")
436                self.kill()
437                await asyncio.sleep(2)
438                return
439            
440            if rover.energy == 0:
441                print(f"{cyan}[{rover.name}] Rover ran out of energy{reset}")
442                self.kill()
443                await asyncio.sleep(2)
444                return
445
446            s_path = len(rover.path)
447
448            if rover.curr >= s_path:
449                rover.curr = s_path - 1            
450
451            next_is_goal = (
452                (rover.status == "returning" or rover.status == "moving") and
453                rover.curr == s_path - 1
454            )
455
456            next_step = rover.path[rover.curr] if not next_is_goal else rover.goal
457            dist_to_next_step = rover.calculate_distance(rover.position, next_step) 
458
459            dx, dy = rover.get_dpos(rover.position, next_step)
460            step_strength = rover.move_step if not next_is_goal else min(rover.move_step, 1)
461            step_size = min(step_strength, dist_to_next_step)
462
463            dx /= dist_to_next_step
464            dy /= dist_to_next_step
465
466            new_pos = (rover.position[0] + dx * step_size, rover.position[1] + dy * step_size)
467
468            rover.energy -= step_size * ENERGY_PER_DISTANCE_UNIT
469
470            cell = rover.map.get_cell(int(new_pos[0]), int(new_pos[1]))
471            if cell and cell.dust_storm:
472                rover.energy -= STORM_COST
473
474            if rover.energy < 0:
475                rover.energy = 0
476
477            sleep_time = step_size / (SIMULATION_SPEED * ROVER_SPEED_UNIT_PER_SEC)
478            await asyncio.sleep(max(0.001, sleep_time))
479
480            collisions = list(filter(
481                lambda collision: not "drone" in collision.id and collision.id != rover.base_jid,
482                rover.world.collides(rover.jid, new_pos)
483            ))
484            s_collisions = len(collisions)
485            if s_collisions > 0:
486                print(f"{CYAN}[{rover.name}] collision detected near {new_pos}, collisions: {collisions}{RESET}")
487                await rover.viz_send_message(f"collision detected! attempting to avoid obstacle")
488
489                alt = await rover.try_go_around(next_step)
490                if alt:
491                    rover.position = alt
492                    print(f"{CYAN}[{rover.name}] avoided obstacle locally.{RESET}")
493                else:
494                    print(f"{CYAN}[{rover.name}] could not avoid locally, crash.{RESET}")
495                    await rover.viz_send_message(f"crash: unable to avoid obstacle")
496                    return
497
498            else:
499                rover.position = new_pos
500
501                distance_after_move = rover.calculate_distance(rover.position, next_step)
502                arrived_next_step = distance_after_move <= COLLISION_RADIUS
503                if arrived_next_step:
504                    rover.position = next_step if not next_is_goal else rover.goal
505                    rover.curr += 1
506                    
507                await rover.viz_update_position(rover.position)
508                await rover.viz_update_battery(100 * rover.energy / MAX_ROVER_CHARGE)
509 
510                print(f"{CYAN}[{rover.name}] {rover.status}... current position: {rover.position}", end = '')
511                print(f", energy: {(100 * rover.energy / MAX_ROVER_CHARGE):.1f}%", end = '')
512                print(f", distance left: {rover.calculate_distance(rover.position, rover.goal)}{RESET}")
513
514                if not arrived_next_step or rover.curr < s_path - 1:
515                    return
516
517                if rover.status == "moving":
518                    rover.status = "arrived"
519
520                    print(f"{CYAN}[{rover.name}] arrived at mission goal {rover.goal}{RESET}")
521                    await rover.viz_send_message(f"arrived at target location {rover.goal}")
522                    await rover.viz_update_status(rover.status)
523
524                    msg = Message(
525                        to=rover.base_jid,
526                        metadata={"performative": "inform", "type": "mission_complete"},
527                        body=str({"position": rover.position})
528                    )
529                    await self.send(msg)
530
531                    rover.add_behaviour(rover.AnalyzeSoil())
532
533                    rover.goal = rover.base_position
534                    old_path = rover.path[::-1]
535
536                    _ = await rover.find_path()
537                    if not rover.path:
538                        print("Using old path to return to base")
539                        rover.path = old_path
540                        rover.goal = rover.base_position
541
542                    rover.status = "returning"
543                    self.kill()
544                    rover.add_behaviour(rover.MoveAlongPath())
545
546                elif rover.status == "returning":
547                    rover.status = "idle"
548                    rover.path = []
549                    rover.goal = None
550                    rover.is_on_base = True
551

568    class AnalyzeSoil(CyclicBehaviour):
569        """
570        Behaviour that simulates a one-time soil/resource analysis at the rover's
571        current location. It samples from defined probabilities and informs the
572        base if resources are discovered.
573        """
574        async def run(self):
575            """
576            Execute resource sampling based on resource_probs. Sends an inform
577            message to the base containing any discovered resources and then
578            terminates the behaviour.
579            """
580            rover = self.agent
581            found_resources = []
582
583            for resource, prob in rover.resource_probs.items():
584                if random.random() < prob:
585                    found_resources.append(resource)
586
587            if found_resources:
588                msg = Message(
589                    to=rover.base_jid,
590                    metadata={"performative": "inform", "type": "resources_found"},
591                    body=str({"position": rover.position, "resources": found_resources})
592                )

174    def calculate_distance(self, pos1: Tuple[float, float], pos2: Tuple[float, float]) -> float:
175        """
176        Compute Euclidean distance between two points.
177
178        Args:
179            pos1 (Tuple[float, float]): First point.
180            pos2 (Tuple[float, float]): Second point.
181
182        Returns:
183            float: Euclidean distance.
184        """
185        return sqrt((pos1[0] - pos2[0]) ** 2 + (pos1[1] - pos2[1]) ** 2)

187    def calculate_pathfinding_cost(self, start_pos: Tuple[float, float], target_pos: Tuple[float, float]) -> float:
188        """
189        Estimate pathfinding cost using randomized Euclidean distance.
190
191        Args:
192            start_pos (Tuple[float, float]): Starting point.
193            target_pos (Tuple[float, float]): Target point.
194
195        Returns:
196            float: Cost estimate.
197        """
198        euclidean_dist = self.calculate_distance(start_pos, target_pos)
199        return euclidean_dist * random.uniform(1.0, 1.1)

201    def compute_mission_time(self, target_pos: Tuple[float, float]) -> float:
202        """
203        Compute estimated total mission time for bidding.
204
205        Args:
206            target_pos (Tuple[float, float]): Mission target.
207
208        Returns:
209            float: Estimated mission time in seconds.
210        """
211        current_energy = self.energy
212
213        dist_to_target = self.calculate_pathfinding_cost(self.position, target_pos)
214        dist_to_base_after = self.calculate_pathfinding_cost(target_pos, self.position)
215        total_distance = dist_to_target + dist_to_base_after
216
217        energy_required = total_distance * ENERGY_PER_DISTANCE_UNIT
218
219        time_to_charge = 0.0
220        if current_energy < energy_required:
221            energy_needed_to_charge = energy_required - current_energy
222            time_to_charge = energy_needed_to_charge / CHARGE_RATE_ENERGY_PER_SEC
223
224        time_to_travel = total_distance / ROVER_SPEED_UNIT_PER_SEC
225
226        mission_time = time_to_charge + time_to_travel
227
228        return mission_time

276        Behaviour responsible for receiving and handling incoming XMPP messages.
277
278        The behaviour processes bid requests from the base, sends proposals or
279        refusals, accepts/rejects mission confirmations, and updates rover

281        structured Message objects with metadata and a stringified body.
282        """
283        async def run(self):
284            """
285            Wait for messages with a short timeout and handle several
286            performatives:
287
288            - cfp / rover_bid_cfp: evaluate mission and send a proposal or refuse
289            - accept_proposal: start moving toward the accepted target
290            - reject_proposal: unlock and remain idle
291            - inform (return_path_to_base): receive a return path and update state
292
293            The method uses eval() to parse message bodies that were serialized
294            with str(dict). It also forwards human-readable updates to the
295            visualizer.
296            """
297            rover = self.agent
298            msg = await self.receive(timeout=3)
299            if not msg:
300                return
301
302            performative = msg.metadata.get("performative")
303            msg_type = msg.metadata.get("type")
304            sender = str(msg.sender)
305
306            # --------------------------
307            # BASE  ROVER : Bid Request
308            # --------------------------
309            if performative == "cfp" and msg_type == "rover_bid_cfp":
310                target_pos = eval(msg.body)
311                print(f"{CYAN}[{rover.name}] Received Bid request from Base for {target_pos}{RESET}")
312                await rover.viz_send_message(f"Received mission bid request for {target_pos}")
313
314                # Check if rover is free
315                if rover.is_locked_by_bid or rover.status != "idle":
316                    # Already committed  refuse
317                    reply = Message(
318                        to=sender,
319                        metadata={"performative": "refuse", "type": "rover_bid_cfp"},
320                        body=str({"reason": "busy or locked"})
321                    )
322                    await self.send(reply)
323                    print(f"{CYAN}[{rover.name}] REFUSING mission at {target_pos} (busy/locked){RESET}")
324                    await rover.viz_send_message(f"Refused mission at {target_pos} (busy/locked)")
325
326                else:
327                    # Rover is available  propose
328                    rover.is_locked_by_bid = True  # lock it for this bid
329
330                    rover.goal = target_pos
331                    mission_status = await rover.find_path()
332                    if mission_status != "viable":
333                        rover.is_locked_by_bid = False
334                        reply = Message(
335                            to=sender,
336                            metadata={"performative": "refuse", "type": "rover_bid_cfp"},
337                            body=str({"reason": mission_status})
338                        )
339                        await self.send(reply)
340                        print(f"{CYAN}[{rover.name}] REFUSING mission at {target_pos} ({mission_status}){RESET}")
341                        await rover.viz_send_message(f"Refused mission at {target_pos} ({mission_status})")
342                        return
343
344                    estimated_mission_time = rover.compute_mission_time(target_pos)
345                    proposal = {"cost": estimated_mission_time, "rover": str(rover.jid)}
346                    reply = Message(
347                        to=sender,
348                        metadata={"performative": "propose", "type": "rover_bid_cfp"},
349                        body=str(proposal)
350                    )
351                    await self.send(reply)
352                    print(f"{CYAN}[{rover.name}] PROPOSING mission at {target_pos} with cost {estimated_mission_time:.2f}{RESET}")

354
355            # Go to target
356            elif performative == "accept_proposal" and msg_type == "rover_bid_cfp":
357                rover.status = "moving"
358                rover.goal = eval(msg.body)["target"]
359                rover.is_locked_by_bid = False  # Unlock after acceptance
360                rover.is_on_base = False # Out of base
361
362                print(f"{CYAN}[{rover.name}] ACCEPTED mission to {rover.goal}{RESET}")
363                await rover.viz_send_message(f"Mission accepted! Moving to {rover.goal}")
364                rover.add_behaviour(rover.MoveAlongPath())
365
366            # Reject proposal  unlock and stay idle
367            elif performative == "reject_proposal" and msg_type == "rover_bid_cfp":
368                rover.is_locked_by_bid = False  # Unlock the rover
369                rover.status = "idle"
370                rover.goal = None
371                rover.path = []
372                print(f"{CYAN}[{rover.name}] REJECTED for mission at {eval(msg.body)['target']}{RESET}")
373                await rover.viz_send_message(f"Bid rejected for mission at {eval(msg.body)['target']}")
374
375            # --------------------------
376            # Inform on rover stats
377            # --------------------------
378            elif performative == "inform" and ontology == "return_path_to_base":
379                rover.path = eval(msg.body)
380                rover.goal = rover.path[-1] if rover.path else None
381                rover.status = "returning"
382                rover.is_locked_by_bid = False
383                print(f"[{rover.name}] Received return path ({len(rover.path)} steps) to base.")

413        async def run(self):
414            """
415            Execute a single movement step along the rover's path. The method
416            performs checks for valid goal/state/path, computes the next
417            position, updates energy, handles collisions, and triggers
418            subsequent behaviours when the rover arrives at its destination or
419            returns to base.
420            """
421            rover = self.agent
422
423            if rover.goal == None:
424                print(f"{CYAN}[{rover.name}] No goal set, cancelling MoveAlongPath{RESET}")
425                self.kill()
426                await asyncio.sleep(2)
427                return
428
429            if rover.status not in ["moving", "returning"]:
430                print(f"{CYAN}[{rover.name}] Doing another task, cancelling MoveAlongPath{RESET}")
431                self.kill()
432                await asyncio.sleep(2)
433                return
434
435            if not rover.path:
436                print(f"{CYAN}[{rover.name}] could not compute path to goal {rover.goal}{RESET}")
437                self.kill()
438                await asyncio.sleep(2)
439                return
440            
441            if rover.energy == 0:
442                print(f"{cyan}[{rover.name}] Rover ran out of energy{reset}")
443                self.kill()
444                await asyncio.sleep(2)
445                return
446
447            s_path = len(rover.path)
448
449            if rover.curr >= s_path:
450                rover.curr = s_path - 1            
451
452            next_is_goal = (
453                (rover.status == "returning" or rover.status == "moving") and
454                rover.curr == s_path - 1
455            )
456
457            next_step = rover.path[rover.curr] if not next_is_goal else rover.goal
458            dist_to_next_step = rover.calculate_distance(rover.position, next_step) 
459
460            dx, dy = rover.get_dpos(rover.position, next_step)
461            step_strength = rover.move_step if not next_is_goal else min(rover.move_step, 1)
462            step_size = min(step_strength, dist_to_next_step)
463
464            dx /= dist_to_next_step
465            dy /= dist_to_next_step
466
467            new_pos = (rover.position[0] + dx * step_size, rover.position[1] + dy * step_size)
468
469            rover.energy -= step_size * ENERGY_PER_DISTANCE_UNIT
470
471            cell = rover.map.get_cell(int(new_pos[0]), int(new_pos[1]))
472            if cell and cell.dust_storm:
473                rover.energy -= STORM_COST
474
475            if rover.energy < 0:
476                rover.energy = 0
477
478            sleep_time = step_size / (SIMULATION_SPEED * ROVER_SPEED_UNIT_PER_SEC)
479            await asyncio.sleep(max(0.001, sleep_time))
480
481            collisions = list(filter(
482                lambda collision: not "drone" in collision.id and collision.id != rover.base_jid,
483                rover.world.collides(rover.jid, new_pos)
484            ))
485            s_collisions = len(collisions)
486            if s_collisions > 0:
487                print(f"{CYAN}[{rover.name}] collision detected near {new_pos}, collisions: {collisions}{RESET}")
488                await rover.viz_send_message(f"collision detected! attempting to avoid obstacle")
489
490                alt = await rover.try_go_around(next_step)
491                if alt:
492                    rover.position = alt
493                    print(f"{CYAN}[{rover.name}] avoided obstacle locally.{RESET}")
494                else:
495                    print(f"{CYAN}[{rover.name}] could not avoid locally, crash.{RESET}")
496                    await rover.viz_send_message(f"crash: unable to avoid obstacle")
497                    return
498
499            else:
500                rover.position = new_pos
501
502                distance_after_move = rover.calculate_distance(rover.position, next_step)
503                arrived_next_step = distance_after_move <= COLLISION_RADIUS
504                if arrived_next_step:
505                    rover.position = next_step if not next_is_goal else rover.goal
506                    rover.curr += 1
507                    
508                await rover.viz_update_position(rover.position)
509                await rover.viz_update_battery(100 * rover.energy / MAX_ROVER_CHARGE)
510 
511                print(f"{CYAN}[{rover.name}] {rover.status}... current position: {rover.position}", end = '')
512                print(f", energy: {(100 * rover.energy / MAX_ROVER_CHARGE):.1f}%", end = '')
513                print(f", distance left: {rover.calculate_distance(rover.position, rover.goal)}{RESET}")
514
515                if not arrived_next_step or rover.curr < s_path - 1:
516                    return
517
518                if rover.status == "moving":
519                    rover.status = "arrived"
520
521                    print(f"{CYAN}[{rover.name}] arrived at mission goal {rover.goal}{RESET}")
522                    await rover.viz_send_message(f"arrived at target location {rover.goal}")
523                    await rover.viz_update_status(rover.status)

370                rover.goal = None

377            # --------------------------

Wait for messages with a short timeout and handle several performatives:

  • cfp / rover_bid_cfp: evaluate mission and send a proposal or refuse
  • accept_proposal: start moving toward the accepted target
  • reject_proposal: unlock and remain idle
  • inform (return_path_to_base): receive a return path and update state
