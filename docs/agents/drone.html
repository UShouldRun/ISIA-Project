 95    class ScanTerrain(CyclicBehaviour):
 96        """
 97        Continuously scan terrain in a grid pattern to detect areas of interest.
 98        
 99        Moves scanning window across the map and probabilistically identifies
100        locations requiring exploration. Initiates mission requests when areas
101        are detected.
102        """
103        
104        async def on_start(self):
105            """
106            Initialize terrain scanning and announce start.
107            """
108            print(f"{GREEN}[{self.agent.name}] Starting terrain scanning...{RESET}")
109            await self.agent.viz_send_message("Starting terrain scanning")
110
111        def in_scan_radius(self, scan_pos: Tuple[int, int]):
112            drone = self.agent
113            return ((drone.position[0] - scan_pos[0]) ** 2 + (drone.position[1] - scan_pos[1]) ** 2) <= drone.scan_radius ** 2
114
115        def is_area_of_interest(self) -> bool:
116            """
117            Determine if current scan position is an area of interest.
118            
119            Returns:
120                bool: True with 25% probability, False otherwise.
121            """
122            return random.random() < 0.25
123
124        async def run(self):
125            """
126            Execute one scanning iteration.
127            
128            Advances scan position, checks for areas of interest, and initiates
129            mission negotiation with available bases. Handles base unavailability
130            by scheduling recheck behavior.
131            """
132            drone = self.agent
133
134            rate = drone.map.length // SCAN_MAP_SIZE
135            drone.current_scan_position[0] += rate
136            if drone.current_scan_position[0] > drone.map.length:
137                drone.current_scan_position[0] = 0
138                drone.current_scan_position[1] += rate
139
140            if drone.current_scan_position[1] > drone.map.height:
141                drone.current_scan_position[1] = 0
142
143            scan_pos = tuple(drone.current_scan_position)
144
145            print(f"{GREEN}[{drone.name}] Scanning area: {scan_pos}{RESET}")
146            
147            if self.is_area_of_interest() and self.in_scan_radius(scan_pos):
148                drone.areas_of_interest.append(scan_pos)
149                print(f"{GREEN}[{drone.name}] Area of interest detected at {scan_pos}{RESET}")
150                await drone.viz_send_message(f"Area of interest detected at {scan_pos}")
151                
152                if not drone.bases:
153                    print(f"{GREEN}[{drone.name}] No bases available. Trying later...{RESET}")
154                    await drone.viz_send_message("No bases available - will retry later")
155                    self.kill()
156                    drone.add_behaviour(drone.RecheckBaseAvailability())
157                    return
158
159                self.kill()
160                drone.add_behaviour(drone.RequestAgentForMission(scan_pos))

162    class RequestAgentForMission(OneShotBehaviour):
163        """
164        Implement FIPA Contract Net Protocol initiator for mission assignment.
165        
166        Sends CFPs to all base stations, collects proposals, selects the best bid,
167        and coordinates mission acceptance/rejection with bases.
168        
169        Attributes:
170            target_position (Tuple[float, float]): Mission target coordinates.
171        """
172        
173        def __init__(self, target_position: Tuple[float, float]):
174            """
175            Initialize mission request behavior.
176            
177            Args:
178                target_position (Tuple[float, float]): Target (x, y) coordinates for exploration.
179            """
180            super().__init__()
181            self.target_position = target_position
182
183        async def run(self):
184            """
185            Execute contract net protocol for mission assignment.
186            
187            Broadcasts CFP to all bases, collects proposals within timeout period,
188            processes responses, and selects winning bid. Resumes scanning after
189            negotiation completion.
190            """
191            drone = self.agent
192 
193            for base_jid in drone.bases:
194                msg = Message(to=base_jid)
195                msg.set_metadata("performative", "cfp")
196                msg.set_metadata("type", "rover_mission_cfp")
197                msg.body = str(self.target_position)
198                await self.send(msg)
199                print(f"{GREEN}[{drone.name}] CFP sent to {base_jid} for mission at {self.target_position}{RESET}")
200            
201            await drone.viz_send_message(f"Requesting mission bids for target {self.target_position}")
202
203            timeout = 4
204
205            start_time = asyncio.get_event_loop().time()
206
207            while asyncio.get_event_loop().time() - start_time < timeout:
208                msg = await self.receive(timeout=1)
209                if msg:
210                    perf = msg.metadata.get("performative")
211                    if perf == "propose":
212                        await self.on_propose(msg)
213                    elif perf == "refuse":
214                        self.on_refuse(msg)
215                    elif perf == "not-understood":
216                        self.on_not_understood(msg)
217                    elif perf == "failure":
218                        self.on_failure(msg)
219                    elif perf == "inform":
220                        self.on_inform(msg)
221
222            await self.on_all_responses_received()
223            drone.add_behaviour(drone.ScanTerrain())
224            
225        def on_refuse(self, message: Message):
226            """
227            Handle refusal from base unable to provide a rover.
228            
229            Moves base to non-available list if it has no rovers.
230            
231            Args:
232                message (Message): Refusal message containing reason.
233            """
234            drone = self.agent
235            reason = eval(message.body)["reason"]
236            print(f"{GREEN}[{drone.name}] Base {str(message.sender).split('@')[0]} refused to bid for mission at {self.target_position}: reason - {reason}{RESET}")
237            if reason == "no_rovers_available":
238               drone.bases.remove(message.sender) 
239               drone.non_available_bases.append(message.sender)
240
241        def on_inform(self, message: Message):
242            """
243            Handle informational messages from bases.
244            
245            Processes base availability updates and moves bases back to
246            available list when rovers become ready.
247            
248            Args:
249                message (Message): Inform message with status update.
250            """
251            drone = self.agent
252            msg_info = eval(message.body)["inform"]
253            print(f"{GREEN}[{drone.name}] message info {msg_info}{RESET}") 
254
255            if msg_info == "has_rovers_available":
256                print(f"{GREEN}[{drone.name}] Base {str(message.sender).split('@')[0]} informed it has rovers available{RESET}")
257                drone.non_available_bases.remove(message.sender)
258                drone.bases.append(message.sender)
259
260        def on_failure(self, message: Message):
261            """
262            Handle failure during contract negotiation.
263            
264            Args:
265                message (Message): Failure notification from base.
266            """
267            print(f"{GREEN}[{self.agent.name}] Base {str(message.sender).split('@')[0]} failed during the contract net protocol.{RESET}")
268
269        def on_not_understood(self, message: Message):
270            """
271            Handle not-understood messages when CFP format is invalid.
272            
273            Args:
274                message (Message): Not-understood notification from base.
275            """
276            print(f"{GREEN}[{self.agent.name}] Base {str(message.sender).split('@')[0]} did not understand the CFP.{RESET}")
277
278        async def on_propose(self, message: Message):
279            """
280            Process proposal bids from base stations.
281            
282            Validates bid format, extracts cost and rover information, and
283            stores valid proposals for later evaluation.
284            
285            Args:
286                message (Message): Proposal message containing bid details.
287            """
288            try:
289                bid_data = eval(message.body)
290                cost = float(bid_data.get("cost", float('inf')))
291                base_jid = bid_data.get("base")
292                rover_jid = bid_data.get("rover")
293                
294                print(f"{GREEN}[{self.agent.name}] Received PROPOSAL from base: {base_jid}: Cost={cost}, Rover={rover_jid}{RESET}")
295                await self.agent.viz_send_message(f"Received bid from {str(base_jid).split('@')[0]} (cost: {cost:.1f}s)")
296                                
297                if base_jid is not None and rover_jid is not None:
298                    self.agent.proposals[base_jid] = {"cost": cost, "base": base_jid, "rover": rover_jid, "proposal_msg": message}
299                else:
300                    print(f"{GREEN}[{self.agent.name}] Ignoring invalid proposal from {message.sender}: No 'rover' specified.{RESET}")
301            
302            except (SyntaxError, TypeError, ValueError):
303                print(f"{GREEN}[{self.agent.name}] Invalid proposal format from {message.sender}. Body: {message.body}{RESET}")
304
305        async def on_all_responses_received(self):
306            """
307            Evaluate all proposals and select winning bid.
308            
309            Chooses base with lowest mission cost, sends acceptance to winner,
310            sends rejections to other bidders, and clears proposal storage.
311            Handles case where no valid proposals are received.
312            """
313            drone = self.agent
314            print(f"{GREEN}[{drone.name}] All responses received for mission at {self.target_position}.{RESET}")
315
316            if not drone.proposals:
317                print(f"{GREEN}[{drone.name}] No proposals received for mission at {self.target_position}. Retrying later.{RESET}")
318                await drone.viz_send_message(f"No bids received for {self.target_position} - will retry")
319                drone.proposals = {} 
320                return
321
322            best_base, best_data = min(drone.proposals.items(), key=lambda x: x[1]["cost"])           
323            min_cost = best_data['cost']
324
325            if best_base:
326                print(f"{GREEN}[{drone.name}] Accepting proposal from {best_base} with cost {min_cost}{RESET}")
327                await drone.viz_send_message(f"Accepted bid from {str(best_base).split('@')[0]} for mission to {self.target_position}")
328
329                accept_msg = Message(to=best_base)
330                accept_msg.set_metadata("performative", "accept_proposal")
331                accept_msg.set_metadata("type", "rover_bid_accepted")
332                accept_msg.body = str({"target": self.target_position, "rover": best_data["rover"]})
333
334                await self.send(accept_msg)
335
336                for base_jid, data in drone.proposals.items():
337                    if base_jid != best_base:
338                        print(f"{GREEN}[{drone.name}] Rejecting proposal from {base_jid}{RESET}")
339                        reject_msg = Message(to=base_jid)
340                        reject_msg.set_metadata("performative", "reject_proposal")
341                        reject_msg.set_metadata("type", "rover_bid_rejected")
342                        reject_msg.body = str({"target": self.target_position, "rover": data["rover"]})
343
344                        await self.send(reject_msg)
345            else:
346                print(f"{GREEN}[{drone.name}] No suitable proposals received for mission at {self.target_position}. Retrying later.{RESET}")
347                await drone.viz_send_message(f"No suitable bids for {self.target_position} - will retry")
348
349            drone.proposals = {}
350
351        async def on_inform(self, message: Message):
352            """
353            Handle mission confirmation from winning base.
354            
355            Receives acknowledgment that rover has been assigned to the mission.
356            
357            Args:
358                message (Message): Inform message with mission details.
359            """
360            sender = str(message.sender)
361            print(f"{GREEN}[{self.agent.name}] Received INFORM from winning base {sender}.{RESET}")

396        """
397        Periodically restore unavailable bases to retry mission assignments.
398        
399        After a delay, moves all bases from non-available back to available list
400        and resumes terrain scanning.
401        """
402        
403        async def run(self):
404            """
405            Wait 30 seconds then restore all bases and resume scanning.
406            
407            Moves bases from non_available_bases back to bases list and
408            restarts the terrain scanning behavior.
409            """
410            drone = self.agent
411            await asyncio.sleep(30)
412
413            print(f"{GREEN}[{drone.name}] preparing to check base availability{RESET}")
414            await drone.viz_send_message("Rechecking base availability")
415            drone.bases.extend(drone.non_available_bases)
416            drone.non_available_bases = []
417
418            drone.add_behaviour(drone.ScanTerrain())
419            self.kill()
