 14class VisualizationServer:
 15    """
 16    WebSocket server responsible for streaming map data, agent updates,
 17    logs, and environmental events to the visualization frontend.
 18
 19    This server manages connected WebSocket clients, broadcasts updates,
 20    and exposes a command channel for frontend-to-agent communication.
 21    """
 22    def __init__(self):
 23        """Initialize the WebSocket app, routes, state containers, and sync events."""
 24        self.app = web.Application()
 25        self.clients = set()
 26        self.app.router.add_get('/ws', self.websocket_handler)
 27        self.client_connected = asyncio.Event()
 28        self.map_data = []
 29
 30    async def websocket_handler(self, request):
 31        """
 32        Handle incoming WebSocket connections, process messages, and
 33        automatically send the full map when a client connects.
 34
 35        Args:
 36            request: aiohttp request instance for the WS upgrade.
 37
 38        Returns:
 39            WebSocketResponse: active WebSocket channel.
 40        """
 41        ws = web.WebSocketResponse()
 42        await ws.prepare(request)
 43
 44        self.clients.add(ws)
 45        self.client_connected.set()
 46        print(f"Client connected. Total clients: {len(self.clients)}")
 47
 48        await self.send_full_map(ws)
 49
 50        try:
 51            async for msg in ws:
 52                if msg.type == aiohttp.WSMsgType.TEXT:
 53                    data = json.loads(msg.data)
 54                    await self.handle_command(data)
 55                elif msg.type == aiohttp.WSMsgType.ERROR:
 56                    print(f'WebSocket error: {ws.exception()}')
 57        finally:
 58            self.clients.discard(ws)
 59            print(f"Client disconnected. Total clients: {len(self.clients)}")
 60            self.client_connected.clear()
 61        return ws
 62
 63    async def handle_command(self, data: str):
 64        """
 65        Process incoming commands from the visualization interface.
 66
 67        Args:
 68            data (str): Parsed JSON payload sent by the frontend.
 69        """
 70        print(f"Received command: {data}")
 71
 72    async def broadcast(self, message: str):
 73        """
 74        Broadcast a JSON-serializable message to all connected clients.
 75
 76        Args:
 77            message (str): The JSON message to send.
 78        """
 79        if not self.clients:
 80            print("No Client found")
 81            return
 82
 83        await asyncio.gather(
 84            *[client.send_json(message) for client in self.clients],
 85            return_exceptions=True
 86        )
 87
 88    async def start_server(self, host: str = '0.0.0.0', port: int = 8080):
 89        """
 90        Start the aiohttp WebSocket server.
 91
 92        Args:
 93            host (str): Host to bind.
 94            port (int): Port to bind.
 95
 96        Returns:
 97            AppRunner: server runner handle.
 98        """
 99        runner = web.AppRunner(self.app)
100        await runner.setup()
101        site = web.TCPSite(runner, host, port)
102        await site.start()
103        print(f"WebSocket server started on ws://{host}:{port}/ws")
104        return runner
105
106    async def wait_for_client(self, timeout=None):
107        """
108        Block until a client connects or a timeout expires.
109
110        Args:
111            timeout (float | None): Optional timeout duration.
112
113        Returns:
114            bool: True if a client connected, False if timed out.
115        """
116        try:
117            await asyncio.wait_for(self.client_connected.wait(), timeout=timeout)
118            print("[WEBSOCKET] Client connection established")
119            return True
120        except asyncio.TimeoutError:
121            print("[WEBSOCKET] Timeout waiting for client connection")
122            return False
123
124    def initialize_map(self, world_map: Map):
125        """
126        Convert the map grid into a flat list of serializable dicts and
127        cache them for streaming to visualization clients.
128
129        Args:
130            world_map (Map): Simulation world map.
131        """
132        flat_map = []
133        max_viz_x = min(world_map.columns, 100)
134        max_viz_y = min(world_map.rows, 100)
135
136        for x in range(max_viz_x):
137            for y in range(max_viz_y):
138                cell = world_map.grid[x][y]
139                flat_map.append(cell.to_dict())
140
141        self.map_data = flat_map
142        self.map_initialized = True
143        print(f"Server map data initialized from simulation world: {len(flat_map)} cells.")
144

146        """
147        Send the full map snapshot to a specific WebSocket client.

149        Args:
150            ws: WebSocketResponse instance.
151        """
152        await ws.send_json({
153            "type": "full_map_init",
154            "map_cells": self.map_data
155        })

237class VisualizationMixin:
238    """
239    Mixin adding visualization capabilities to SPADE agents.
240
241    Stores tracked attributes (position, battery, status, etc.) and provides
242    helper methods to send structured updates to the visualization server.
243    """
244    def setup_visualization(self, viz_server, agent_type: str, agent_jid: str, position: Tuple[float, float], battery: float = 100.0, color: Optional[str] = None, radius: int = 5):
245        """
246        Initialize visualization properties for the agent.
247
248        Args:
249            viz_server (VisualizationServer): The visualization backend.
250            agent_type (str): Category of agent ('rover', 'drone', etc.).
251            agent_jid (str): Agent identifier.
252            position (Tuple[float, float]): Initial coordinates.
253            battery (float): Initial battery level.
254            color (str | None): Custom color.
255            radius (int): Rendering radius.
256        """
257        self.viz_server = viz_server
258        self.agent_jid = agent_jid
259        self.viz_agent_type = agent_type
260        self.viz_color = color
261        self.viz_radius = radius
262        self.viz_position = position
263        self.viz_battery = battery
264        self.viz_status = "initializing"
265
266    async def viz_update_position(self, pos: Tuple[float, float]):
267        """Update agent position in visualization."""
268        self.viz_position = (float(pos[0]), float(pos[1]))
269
270    async def viz_update_battery(self, battery: float = 100.0):
271        """Update agent battery level."""
272        self.viz_battery = float(battery)
273
274    async def viz_update_status(self, status: str):
275        """Update agent status string."""
276        self.viz_status = status
277
278    async def viz_send_update(self):
279        """Push the agent's complete state update to the server."""
280        if hasattr(self, 'viz_server') and self.viz_server:
281            await self.viz_server.send_agent_update(
282                agent_id=self.agent_jid.split("@")[0],
283                agent_type=self.viz_agent_type,
284                x=self.viz_position[0],
285                y=self.viz_position[1],
286                battery=self.viz_battery,
287                status=self.viz_status,
288                color=self.viz_color,
289                radius=self.viz_radius
290            )
291
292    async def viz_report_resource(self, resource_id: str, x: float, y: float):
293        """Report a discovered resource to the visualization layer."""
294        if hasattr(self, 'viz_server') and self.viz_server:
295            await self.viz_server.send_resource_discovered(resource_id, float(x), float(y))
296            await self.viz_server.send_message(
297                sender=self.agent_jid.split("@")[0],
298                content=f"Discovered {resource_id} at ({x:.1f}, {y:.1f})"
299            )
300
301    async def viz_report_hazard(self, hazard_id: str, x: float, y: float, radius: float = 5):
302        """Report a detected hazard at a location."""
303        if hasattr(self, 'viz_server') and self.viz_server:
304            await self.viz_server.send_hazard_detected(hazard_id, float(x), float(y), float(radius))
305            await self.viz_server.send_message(
306                sender=self.agent_jid.split("@")[0],
307                content=f"Hazard detected: {hazard_id} at ({x:.1f}, {y:.1f})"
308            )
309
310    async def viz_mark_explored(self, x: float, y: float):
311        """Mark a cell as explored visually."""
312        if hasattr(self, 'viz_server') and self.viz_server:
313            await self.viz_server.send_cell_explored(int(x), int(y))
314
315    async def viz_send_message(self, content: str):
316        """Send a log message to visualization."""
317        if hasattr(self, 'viz_server') and self.viz_server:
318            await self.viz_server.send_message(
319                sender=self.agent_jid.split("@")[0],
320                content=content
321            )

327        if hasattr(self.agent, 'viz_server'):
