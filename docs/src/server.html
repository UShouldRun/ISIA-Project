 67def random_pos_in_base(world: World, base_name: str, base_centers: Dict[str, Tuple[float, float]], base_radii: Dict[str, float]):
 68    """
 69    Generate a random collision-free position within a specified base radius.
 70
 71    Args:
 72        world (World): The world object containing all existing objects.
 73        base_name (str): Name of the base to generate the position within.
 74        base_centers (Dict[str, Tuple[float, float]]): Dictionary mapping base names to their centers.
 75        base_radii (Dict[str, float]): Dictionary mapping base names to their radii.
 76
 77    Returns:
 78        Tuple[float, float]: Random collision-free coordinates within the base.
 79    """
 80    if base_name not in base_centers:
 81        print(f"[WARNING] Base '{base_name}' not found, using default position")
 82        return (100, 100)
 83    
 84    base_center = base_centers[base_name]
 85    base_radius = base_radii.get(base_name, 50)
 86    
 87    max_attempts = 100
 88    for _ in range(max_attempts):
 89        x = random.uniform(base_center[0] - 0.15 * base_radius, base_center[0] + 0.15 * base_radius)
 90        y = random.uniform(base_center[1] - 0.15 * base_radius, base_center[1] + 0.15 * base_radius)
 91        if all(((x - o.pos[0]) ** 2 + (y - o.pos[1]) ** 2) ** 0.5 > COLLISION_RADIUS for o in world.objects):
 92            return (x, y)
 93    
 94    print("[WARNING] Could not find collision-free position, using fallback")
 95    return (base_center[0], base_center[1])

 97def generate_world(config: Dict[str, Any], tag: str) -> Tuple[World, Map, Dict[str, Tuple[float, float]], List[Tuple[float, float]]]:
 98    """
 99    Generate the world environment, map, and initial positions of rovers and drones.
100
101    Args:
102        config (Dict[str, Any]): Simulation configuration.
103        tag (str): Tag used to construct agent JIDs.
104
105    Returns:
106        Tuple[World, Map, Dict[str, Tuple[float, float]], List[Tuple[float, float]], List[Tuple[float, float]]]:
107        - World object containing all agents and objects.
108        - Map object representing terrain.
109        - Dictionary of base centers by name.
110        - List of rover positions.
111        - List of drone positions.
112    """
113    world_config = config.get("world", {})
114    base_configs = config.get("bases", [])
115    rover_configs = config.get("rovers", [])
116    drone_configs = config.get("drones", [])
117    
118    map_limit = tuple(world_config.get("map_limit", [100, 100]))
119    
120    world_map = Map(map_limit)
121    world = World([])
122    
123    base_centers = {}
124    base_radii = {}
125    
126    # --- Create base objects ---
127    for base_config in base_configs:
128        base_name = base_config.get("name", base_config["jid"])
129        base_center = tuple(base_config.get("center", [100, 100]))
130        base_radius = base_config.get("radius", 50)
131        
132        base_centers[base_name] = base_center
133        base_radii[base_name] = base_radius
134        world.objects.append(WorldObject(f"{base_config['jid']}@{tag}", base_center))
135        
136    # --- Process rover positions ---
137    rover_positions = []
138    for rover_config in rover_configs:
139        pos = rover_config.get("position", "random_in_base")
140        if pos == "random_in_base":
141            rover_base = rover_config.get("base")
142            if not rover_base and base_configs:
143                rover_base = base_configs[0].get("name", base_configs[0]["jid"])
144            pos = random_pos_in_base(world, rover_base, base_centers, base_radii)
145        else:
146            pos = tuple(pos)
147        rover_positions.append(pos)
148    
149    # --- Register rover world objects ---
150    for i, (rover_config, pos) in enumerate(zip(rover_configs, rover_positions), start=1):
151        rover_name = rover_config.get("name", f"rover{i}")
152        world.objects.append(WorldObject(f"{rover_config['jid']}@{tag}", pos))
153    
154    # --- Process drone positions ---
155    drone_positions = []
156    for drone_config in drone_configs:
157        pos = tuple(drone_config.get("position", [500, 500]))
158        drone_positions.append(pos)
159        drone_name = drone_config.get("name", drone_config["jid"])
160        world.objects.append(WorldObject(f"{drone_config['jid']}@{tag}", pos))
161    
162    return world, world_map, base_centers, rover_positions, drone_positions

234            await viz_server.send_map_updates(flat_map)

316        """
317        Process incoming commands from the visualization interface.
318
319        Args:

376        """
377        Starts the simulation from a config file.
378        This is basically your current `main()` function, refactored to run on demand.
379        """
380        started_agents = []
381        hazard_task = None
382        
383        try:
384            self.simulation_running = True
385            config = load_config(config_path)
386
387            # --- EXTRACT CONFIG VALUES ---
388            sim_config = config.get("simulation", {})
389            duration = sim_config.get("duration_seconds", 600)
390            tag = sim_config.get("tag", "spade")
391
392            base_configs = config.get("bases", [])
393            drone_configs = config.get("drones", [])
394            rover_configs = config.get("rovers", [])
395
396            # --- WORLD INITIALIZATION ---
397            world, world_map, base_centers, rover_positions, drone_positions = generate_world(config, tag)
398            
399            # Update server's world_map reference
400            self.world_map = world_map
401            self.initialize_map(world_map)
402            await self.broadcast({
403                "type": "full_map_init",
404                "map_cells": self.map_data
405            })
406
407            # --- SETUP LOGGING ---
408            setup_logging(config)
409
410            # --- BUILD JID MAPS ---
411            base_jids = {b["jid"]: f"{b['jid']}@{tag}" for b in base_configs}
412            drone_jids = {d["jid"]: f"{d['jid']}@{tag}" for d in drone_configs}
413            rover_jids = {r["jid"]: f"{r['jid']}@{tag}" for r in rover_configs}
414
415            # --- CREATE AGENTS ---
416            agents_to_start = []
417
418            # Create Bases
419            bases = {}
420            for base_config in base_configs:
421                base_jid = base_jids[base_config["jid"]]
422                base_name = base_config.get("name", base_config["jid"])
423                base_center = base_centers[base_name]
424
425                rovers_for_this_base = [
426                    rover_jids[r["jid"]] 
427                    for r in rover_configs 
428                    if r.get("base", base_configs[0].get("name") if base_configs else None) == base_name
429                ]
430                drones_for_this_base = [drone_jids[d["jid"]] for d in drone_configs]
431
432                base = Base(
433                    base_jid,
434                    base_name,
435                    base_center,
436                    rovers_for_this_base,
437                    drones_for_this_base,
438                    radius=base_config.get("radius", 50),
439                    viz_server=self
440                )
441                bases[base_jid] = base
442                agents_to_start.append(base)
443
444            # Create Drones
445            drones = {}
446            for drone_config, drone_pos in zip(drone_configs, drone_positions):
447                drone_jid = drone_jids[drone_config["jid"]]
448                known_base_names = drone_config.get("known_bases", [])
449                known_base_jids = [base_jids[b] for b in known_base_names if b in base_jids]
450
451                drone = Drone(
452                    drone_jid,
453                    drone_config.get("name", drone_config["jid"]),
454                    world,
455                    world_map,
456                    drone_pos,
457                    known_bases=known_base_jids,
458                    scan_radius=drone_config.get("scan_radius", 20.0),
459                    viz_server=self
460                )
461                drones[drone_jid] = drone
462                agents_to_start.append(drone)
463
464            # Create Rovers
465            rovers = {}
466            for rover_config, rover_pos in zip(rover_configs, rover_positions):
467                rover_jid = rover_jids[rover_config["jid"]]
468                rover_base_name = rover_config.get("base", base_configs[0].get("name") if base_configs else None)
469                rover_base_jid = base_jids.get(rover_base_name)
470
471                rover = Rover(
472                    rover_jid,
473                    rover_config.get("name", rover_config["jid"]),
474                    rover_pos,
475                    world,
476                    world_map,
477                    base_jid=rover_base_jid,
478                    base_radius=bases[rover_base_jid].radius if rover_base_jid else 50,
479                    viz_server=self
480                )
481                rovers[rover_jid] = rover
482                agents_to_start.append(rover)
483
484            # --- START AGENTS ---
485            for agent in agents_to_start:
486                await agent.start()
487                started_agents.append(agent)
488            
489            await self.send_message("server", "[INFO] All agents started")
490
491            # Start hazards task
492            hazard_task = asyncio.create_task(simulate_hazards(world_map, self, interval=10))
493
494            # --- RUN SIMULATION ---
495            try:
496                start_time = asyncio.get_event_loop().time()
497                while asyncio.get_event_loop().time() - start_time < duration:
498                    await self.pause_event.wait()  # Wait if paused
499                    await asyncio.sleep(1)  # Check every second
500            except asyncio.CancelledError:
501                logging.info("[SIMULATION] Cancel received, stopping simulation immediately.")
502                raise  # propagate so the finally block handles cleanup
503
504
505            # Normal completion
506            await self.send_message("server", "[INFO] Simulation completed.")
507            await self.broadcast({"type": "simulation_completed"})
508            
509        except asyncio.CancelledError:
510            # Simulation was stopped/reset
511            await self.send_message("server", "[INFO] Simulation stopped")
512            await self.broadcast({"type": "simulation_stopped"})
513            raise  # Re-raise to properly handle cancellation
514            
515        except FileNotFoundError as e:
516            await self.send_message("server", f"[ERROR] Config file not found: {config_path}")
517            await self.broadcast({"type": "error", "message": f"Config file not found: {config_path}"})
518        except Exception as e:
519            logging.exception("Error during simulation")
520            await self.send_message("server", f"[ERROR] Simulation failed: {str(e)}")
521            await self.broadcast({"type": "error", "message": str(e)})
522        finally:
523            # CLEANUP - Always stop agents and hazard task
524            if hazard_task:
525                hazard_task.cancel()
526                try:
527                    await hazard_task
528                except asyncio.CancelledError:
529                    pass
530                    
531            for agent in started_agents:
532                try:
533                    await agent.stop()
534                except Exception as e:
535                    logging.error(f"Error stopping agent: {e}")
536            
537            self.simulation_running = False
538            self.simulation_paused = False
539            self.pause_event.set()  # Reset pause state
540            await self.send_message("server", "[INFO] All agents stopped. Ready for new simulation.")
